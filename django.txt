***V.environment içerisine proje yüklemek için: "django-admin startproject projectName".

14-API "department" ve "employee" methodları için "EmployeeApp/views" gir ve şöyle yap:
"from django.shortcuts import render
from django.views.decorators.csrf import csrf_exempt
from rest_framework.parsers import JSONParser
from django.http.response import JsonResponse
from EmployeeApp.models import Departments, Employees
from EmployeeApp.serializers import DepartmentSerializer, EmployeeSerializer
@csrf_exempt
def departmentApi(request, id=0):
    if request.method == 'GET':
        return JsonResponse(DepartmentSerializer(Departments.objects.all(), many=True).data, safe=False)
    elif request.method == 'POST':
        department_serializer = DepartmentSerializer(
            data=JSONParser().parse(request))
        if department_serializer.is_valid():
            department_serializer.save()
            return JsonResponse('Added Successfully!', safe=False)
        return JsonResponse('Failed to Add!', safe=False)
    elif request.method == 'PUT':
        department_data = JSONParser().parse(request)
        department = Departments.objects.get(
            DepartmentId=department_data['DepartmentId'])
        department_serializer = DepartmentSerializer(
            department, data=department_data)
        if department_serializer.is_valid():
            department_serializer.save()
            return JsonResponse('Updated Successfully!!', safe=False)
        return JsonResponse('Failed to Update.', safe=False)
    elif request.method == 'DELETE':
        Departments.objects.get(DepartmentId=id).delete()
        return JsonResponse('Deleted Succeffully!!', safe=False)
@csrf_exempt
def employeeApi(request, id=0):
    if request.method == 'GET':
        return JsonResponse(EmployeeSerializer(Employees.objects.all(), many=True).data, safe=False)
    elif request.method == 'POST':
        employee_serializer = EmployeeSerializer(
            data=JSONParser().parse(request))
        if employee_serializer.is_valid():
            employee_serializer.save()
            return JsonResponse('Added Successfully!', safe=False)
        return JsonResponse('Failed to Add!', safe=False)
    elif request.method == 'PUT':
        employee_data = JSONParser().parse(request)
        employee = Employees.objects.get(
            EmployeeId=employee_data['EmployeeId'])
        employee_serializer = EmployeeSerializer(employee, data=employee_data)
        if employee_serializer.is_valid():
            employee_serializer.save()
            return JsonResponse('Updated Successfully!!', safe=False)
        return JsonResponse('Failed to Update.', safe=False)
    elif request.method == 'DELETE':
        Employees.objects.get(EmployeeId=id).delete()
        return JsonResponse('Deleted Succeffully!!', safe=False)".

"from rest_framework import status
from rest_framework.response import Response
from app1.models import Makale
from app1.api.serializers import MakaleSerializer
from rest_framework.views import APIView
from rest_framework.generics import get_object_or_404
class MakaleListCreateApiView(APIView):
    def get(self, request):
        makaleler = Makale.objects.filter(aktif = True)
        serializer = MakaleSerializer(makaleler, many = True)
        return Response(serializer.data)
    def post(self, request):
        serializer = MakaleSerializer(data = request.data)
        if serializer.is_valid():
            serializer.save()
            return Response(serializer.data, status = status.HTTP_201_CREATED)
        return Response(serializer.errors, status = status.HTTP_400_BAD_REQUEST)
class MakaleDetailAPIView(APIView):
    def get_object(self, pk):
        makale_instance = get_object_or_404(Makale,pk=pk)
        return makale_instance
    def get(self, request, pk):
        makale = self.get_object(pk=pk)
        serializer = MakaleSerializer(makale)
        return Response(serializer.data)
    def put(self, request, pk):
        makale = self.get_object(pk=pk)
        serializer = MakaleSerializer(makale, data=request.data)
        if serializer.is_valid():
            serializer.save()
            return Response(serializer.data)
        return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
    def delete(self, request, pk):
        makale = self.get_object(pk=pk)
        makale.delete()
        return Response(status=status.HTTP_204_NO_CONTENT)"

*****"StringRelatedField()" kullanımı:
1-"ForeignKey" veya "OneToOneField" ile ilişkilendirilmiş modelin propertisin aldığı valueyi, ilişkilendirildiği modelin "__str__" metodunun valuesine dönüştürür.Mesela "Makale" classına şöyle "ForeignKey" ata:
"class Makale(models.Model):
yazar = models.ForeignKey(Gazeteci)".
2-Serializer gel ve yeni propertiyi şöyle yaz:
"class MakaleSerializer(serializers.ModelSerializer):
    yazar = serializers.StringRelatedField()".
3-Şimdi ise ilişkilendirilmiş olan "Gazeteci" modeline gel ve şöyle yap:
"class Gazeteci(models.Model):
isim = models.CharField(max_length=120)
soyisim = models.CharField(max_length=120)
def __str__(self):
    return f'{self.isim} __ {self.soyisim}'".
4-Böylece databasede foreign key olan columnlar, "__str__" metodunun valuesini alacaktır.Fakat object post ederken "yazar" propertisi hata verecektir!O yüzden post işlemi varsa, bu metodu kullanmaktan kaçın.

*****"HyperlinkedRelatedField()" kullanımı:
1-İlişkilendirilmiş modellerde mevcut propertye karşılık gelen valuelerin viewlerini, link olarak getirecektir.Mesela "Makale" classına şöyle ekle:
"class Makale(models.Model):
yazar = models.ForeignKey(Gazeteci, related_name='makaleler1')".
2-"urls" gel ve şöyle ekle:
"path('makaleler/<int:pk>', api_views.MakaleDetailAPIView.as_view(), name='makale-detay1')".
2-Yukarıdan "name"i al ve  "Gazeteci" serializera şöyle ekle:
"class GazeteciSerializer(serializers.ModelSerializer):
makaleler1 = serializers.HyperlinkedRelatedField(
    many=True,
    read_only=True,
    view_name='makale-detay1')
class Meta:
    model = Gazeteci
    fields = '__all__'"
3-"many=True" yaptık çünkü bir "Gazeteci"nin, birden fazla makalesi olacaktır ve query set olarak döndürecektir."read_only=True" yaptık çünkü yeni bir "Yazar" eklendiğinde "makaleler1" propertisini post etmemeliyiz.Şimdi "Gazeteci" viewine gel ve şöyle yap:
"class GazeteciListCreateApiView(APIView):

    yazarlar = Gazeteci.objects.all()

4-Artık "Gazeteci" objectleri get ettiğimizde, sahip oldukları makaleleri urller şeklinde göreceğiz.

*****"@classmethod" kullanımı:

*****"inheritance":
1-Referans olarak üretildiği classın bütün properti ve metodlarını kendisinde barındırır.Bu properti ve metodlar, mevcut class içerisinde çağrılarak override edilebilirler.Ayrıca mevcut classa yeni property ve metodlar atanabilir.Mesela:
"class Personel:
    zam_orani = 1.05
    def __init__(self, isim1, soyisim1):
        self.isim = isim1
        self.soyisim = soyisim1
    def tam_isim(self):
        return f'{self.isim} __ {self.soyisim}'
class Yazilimci(Personel):
pass
yazilimci_1 = Yazilimci('Ahmed','Arslan')
print(Yazilimci.zam_orani)
print(yazilimci_1.tam_isim())
print(Yazilimci.tam_isim(yazilimci_1))
print(Personel.tam_isim(yazilimci_1))".
2-"Yazilimci" classını "Personel" classından türettik ve böylece "Personel" classına ait tüm porperti ve metodlara sahip oldu.Fakat "Yazilimci" classında "zam_orani" propertisini override etmedik.Böylece "zam_orani" propertisini print edince, referansı olan "Personel" classının değerini çekti.Ayrıca "yazilimci_1" object instanceının "tam_isim" metodunu 3 farklı şekilde çektik.
3-"zam_orani" propertisini şöyle override edebilirdik:
"class Yazilimci(Personel):
    zam_orani=3".

*****Modelde veri tipleri:
****String:
**Kısa string değerler için mesela "class Makale(models.Model)" classı içerisine:
"yazar = models.CharField(max_length=150)".
**Uzun string değerler için mesela "class Makale(models.Model)" classı içerisine:
"metin = models.TextField()".
***"TextField()" uzun textler için, "CharField()" ise kısa textler için kullanılır.Mesela isim-soy isim, doğum yeri, vs "CharField" şeklindedir.Paragraflar, kompozisyonlar, vs ise "TextField" şeklindedir.
****Integer:
**Sadece pozitif değerler için mesela "class Makale(models.Model)" classı içerisine:
"degerlendirme = models.PositiveIntegerField(
    validators=[MinValueValidator(1),MaxValueValidator(5)])".
****Date:
**Kesin bir date belirtmek için mesela "class Makale(models.Model)" classı içerisine:
"yayımlanma_tarihi = models.DateField()".
**Object yaratıldığı anda otomatik eklenen ve bir daha değişmeyecek olan date belirtmek için:
1-Mesela "class Makale(models.Model)" classı içerisine şöyle yaz:
"yaratilma_tarihi = models.DateTimeField(auto_now_add=True)".
2-Serialize ederken 'read_only=True' parametresiyle şöyle kullan:
"yaratilma_tarihi = serializers.DateTimeField(read_only=True)"
3-Http requestleri için "def update(self, instance, validated_data)" metodunda kullanma! Çünkü zaten model içerisinde kendisi halledecektir.
**Yaratılan objectle alakalı her yeni işlem yapıldığında, o yapılan yeni işlemin tarihini alan bir date belirtmek için:
1-Mesela "class Makale(models.Model)" classı içerisine şöyle yaz:
"güncelleme_tarihi = models.DateTimeField(auto_now=True)".
2-Serialize ederken 'read_only=True' parametresiyle şöyle kullan:
"güncelleme_tarihi = serializers.DateTimeField(read_only=True)".
3-Http requestleri için "def update(self, instance, validated_data)" metodunda kullanma! Çünkü zaten model içerisinde kendisi halledecektir.
****Booleans:
**Boolean bir durum belirtmek için mesela "class Makale(models.Model)" classı içerisine:
"aktif = models.BooleanField()"
**Boolean duruma ise default bir atamayı şöyle yaparız[böylece object oluşturulduğu an bu değeri beraberinde alır]:
"aktif = models.BooleanField(default=True)"

4-Shellde "from app1.api.serializers import MakaleSerializer" yazarak "Makale" modeli için oluşturduğumuz serializeı import et.
5-Artık objecti çekebiliriz.Shellde şöyle bir variable tanımla: "instance1 = Makale.objects.first()".

8-"serializer1.data" yazarsak, objectin tüm propertilerini JSONa benzer formatta görüntüleriz.
9-Eğer JSONa benzer formattaki görüntüyü daha güzelleştirmek istersek, "from pprint import pprint" yaz.
10-"pprint(serializer1.data)" yazarsak objectin propertileri, çok daha anlaşılır görüntülenecektir.

14-Json formatındaki datayı databasee artık kaydedebiliriz.Bunun için önce "import io" ve "from rest_framework.parsers import JSONParser" yaz.
15-Önce bizim "byte" formatındaki datamızı "stream1 = io.BytesIO(data1)" yazarak "stream"e çeviririz.
16-Şimdi "data1 = JSONParser().parse(stream1)" yazarak artık datamızı, pythonın okuyabileceği bir formata dönüştürdük.
17-"serializer1 = MakaleSerializer(data=data1)" yazarak datayı serialize ederiz.Görüntülersek de artık data, açıklamalarıyla beraber görüntülenecektir.
18-Artık serializeımıza "serializer1.is_valid()" yazarak kontrol sağlayabiliriz.
19-Hatta "serializer1.validated_data" yazarak da objectin datalarını bir python veri datası kümesi olarak görüntüleriz.
